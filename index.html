#define _CRT_SECURE_NO_WARNINGS #include #include #include #include // 设置控制台颜色 void SetColor(int color) { SetConsoleTextAttribute(GetStdHandle(STD_OUTPUT_HANDLE), color); } // 心形公式判断 int isInHeart(double x, double y, double zoom) { x /= zoom; y /= zoom; return pow(x * x + y * y - 1, 3) - x * x * pow(y, 3) <= 0; } int main() { char name[50]; // 强制设置控制台中文编码，彻底避免乱码 SetConsoleCP(936); SetConsoleOutputCP(936); system("title 专属爱心"); // 隐藏控制台光标，消除动画闪烁 CONSOLE_CURSOR_INFO cursorInfo = { 1, FALSE }; SetConsoleCursorInfo(GetStdHandle(STD_OUTPUT_HANDLE), &cursorInfo); printf("请输入你想展示的名字："); scanf("%49s", name); system("cls"); int nameLen = strlen(name); // 名字占用的总字符宽度 const int lineCharCount = 60; // 每行固定字符数，和x轴完全对应 double zoom = 1.0; int dir = 1; int colorCycle = 0; // 【关键参数】名字的垂直位置，数值越大越靠上，越小越靠下 // 想要再往上就调大这个数，比如0.3；想往下就调小，比如0.1 const double nameVerticalPos = 0.25; while (1) { // 光标重置到左上角，减少闪烁 COORD pos = { 0, 0 }; SetConsoleCursorPosition(GetStdHandle(STD_OUTPUT_HANDLE), pos); // 渐变颜色组，可自行修改 int colors[] = { 12, 13, 5, 9, 14, 15 }; int currentColor = colors[colorCycle % 6]; SetColor(currentColor); // 逐行绘制爱心 for (double y = 1.3; y > -1.1; y -= 0.1) { // 【核心修改】名字行的位置，从原来的y=0改成y=nameVerticalPos，整体上移 if (fabs(y - nameVerticalPos) < 0.05) { char lineBuffer[100] = { 0 }; int nameStartIndex = (lineCharCount - nameLen) / 2; // 水平绝对居中 // 先生成整行的爱心字符 int charIndex = 0; for (double x = -1.5; x < 1.5; x += 0.05) { lineBuffer[charIndex] = isInHeart(x, y, zoom) ? '*' : ' '; charIndex++; } // 把名字覆盖到居中位置 memcpy(lineBuffer + nameStartIndex, name, nameLen); // 分段打印，实现名字高亮 for (int i = 0; i < nameStartIndex; i++) { printf("%c", lineBuffer[i]); } SetColor(15); // 名字白色高亮 printf("%s", name); SetColor(currentColor); // 恢复爱心颜色 for (int i = nameStartIndex + nameLen; i < lineCharCount; i++) { printf("%c", lineBuffer[i]); } printf("\n"); continue; } // 非名字行，正常绘制爱心 for (double x = -1.5; x < 1.5; x += 0.05) { printf("%c", isInHeart(x, y, zoom) ? '*' : ' '); } printf("\n"); } // 跳动动画控制 zoom += dir * 0.02; if (zoom >= 1.15) dir = -1; if (zoom <= 0.85) dir = 1; // 颜色渐变控制 if (colorCycle % 20 == 0) colorCycle++; colorCycle++; Sleep(50); } return 0; }
